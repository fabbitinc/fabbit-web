export const PaletteRow = ({ name, subtitle, swatches }) => {
  return (
    <div style={tableRow}>
      <div>
        <p style={nameTitle}>{name}</p>
        <p style={nameSub}>{subtitle}</p>
      </div>
      <div>
        <div style={stripWrap}>
          {swatches.map((swatch) => (
            <div
              key={swatch.label}
              style={{
                flex: 1,
                background: swatch.color,
              }}
            />
          ))}
        </div>
        <div style={labelGrid(swatches.length)}>
          {swatches.map((swatch) => (
            <div key={`${swatch.label}-label`} style={labelBox}>
              <div style={{ fontWeight: 600 }}>{swatch.label}</div>
              <div style={codeText}>{swatch.value}</div>
            </div>
          ))}
        </div>
      </div>
    </div>
  );
};

export const ColorChipGrid = ({ colors }) => {
  return (
    <div style={chipGrid}>
      {colors.map((color) => (
        <div key={color} style={chip}>
          <div style={{ ...chipSwatch, background: color }} />
          <div style={chipLabel}>{color}</div>
        </div>
      ))}
    </div>
  );
};
